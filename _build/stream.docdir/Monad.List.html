Module Monad.List


module List: BasePlus  with type 'a m = 'a list
include BatMonad.S
val zero : unit -> 'a m
val plus : 'a m -> 'a m -> 'a m
val null : 'a m -> bool
null x implies that x is zero. If you do not want to or cannot answer whether a given x is zero, then null x should be false.
